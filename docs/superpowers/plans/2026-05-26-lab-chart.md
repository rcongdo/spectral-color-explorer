# Lab Chart Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add a CIE L*a*b* visualization panel (a*b* scatter + L* bar) to the right of the spectral graph that updates on `updateFull()`.

**Architecture:** Single-file HTML/CSS/JS change to `spectral-reflectance.html`. Static scaffold drawn once in `initLabChart()` (called from startup); dynamic dots and ticks cleared and redrawn by `renderLabChart(results)` (called from `updateFull()`). The spectral graph area is wrapped in a new `#spectral-graph-wrap` div so `#graph-legend` stays scoped beneath it, while `#lab-chart` SVG sits beside it in a flex row.

**Tech Stack:** Vanilla HTML/CSS/SVG (no build step, no dependencies). Browser-based test runner in `test.html`.

- [ ] **Step 4: Add `initLabChart()` function**

Add this function after `abToSvg` / `lToSvgY`. It creates all static SVG elements (axes, grid, gradient, achromatic dot, labels) once at startup. `svgEl(tag, attrs)` is already defined in the file — use it.

```javascript
function initLabChart() {
  const svg = document.getElementById('lab-chart');
  // Background (matches spectral graph)
  svg.appendChild(svgEl('rect', { x: 0, y: 0, width: 220, height: 300, fill: '#0a1628' }));

  // ── a*b* scatter sub-area ──────────────────────────────────
  // Grid lines at ±40
  const gridAttrs = { stroke: '#1e293b', 'stroke-width': '1' };
  // a* = ±40 vertical lines
  [{ a: -40, b: 0 }, { a: 40, b: 0 }].forEach(({ a }) => {
    const x = abToSvg(a, 0).x;
    svg.appendChild(svgEl('line', { ...gridAttrs, x1: x, y1: LC_Y0, x2: x, y2: LC_Y0 + LC_PH }));
  });
  // b* = ±40 horizontal lines
  [40, -40].forEach(b => {
    const y = abToSvg(0, b).y;
    svg.appendChild(svgEl('line', { ...gridAttrs, x1: LC_X0, y1: y, x2: LC_X0 + LC_PW, y2: y }));
  });

  // Origin crosshair (brighter)
  const crossAttrs = { stroke: '#334155', 'stroke-width': '1' };
  const ox = abToSvg(0, 0).x;
  const oy = abToSvg(0, 0).y;
  svg.appendChild(svgEl('line', { ...crossAttrs, x1: ox, y1: LC_Y0, x2: ox, y2: LC_Y0 + LC_PH }));
  svg.appendChild(svgEl('line', { ...crossAttrs, x1: LC_X0, y1: oy, x2: LC_X0 + LC_PW, y2: oy }));

  // Axis labels
  const labelAttrs = { fill: '#475569', 'font-size': '9', 'font-family': 'system-ui, sans-serif' };
  // a* label centered below x-axis
  const aLabelEl = svgEl('text', { ...labelAttrs, x: LC_X0 + LC_PW / 2, y: LC_Y0 + LC_PH + 12, 'text-anchor': 'middle' });
  aLabelEl.textContent = 'a*';
  svg.appendChild(aLabelEl);
  // b* label rotated -90° alongside y-axis
  const bLabelEl = svgEl('text', { ...labelAttrs, x: 0, y: 0, 'text-anchor': 'middle',
    transform: `rotate(-90) translate(${-(LC_Y0 + LC_PH / 2)}, 12)` });
  bLabelEl.textContent = 'b*';
  svg.appendChild(bLabelEl);

  // Achromatic reference dot at origin
  svg.appendChild(svgEl('circle', { cx: ox, cy: oy, r: 4, fill: '#334155', stroke: '#475569', 'stroke-width': '1' }));

  // ── L* bar ────────────────────────────────────────────────
  // Gradient definition
  const defs = svgEl('defs', {});
  const grad = svgEl('linearGradient', { id: 'lstar-grad', x1: '0', y1: '1', x2: '0', y2: '0',
    gradientUnits: 'objectBoundingBox' });
  const stop0 = svgEl('stop', { offset: '0%', 'stop-color': '#000000' });
  const stop1 = svgEl('stop', { offset: '100%', 'stop-color': '#ffffff' });
  grad.appendChild(stop0);
  grad.appendChild(stop1);
  defs.appendChild(grad);
  svg.appendChild(defs);

  // Bar rectangle
  svg.appendChild(svgEl('rect', { x: LC_LX, y: LC_LY, width: LC_LW, height: LC_LH,
    fill: 'url(#lstar-grad)', stroke: '#1e293b', 'stroke-width': '1' }));

  // L* label above bar
  const lLabelEl = svgEl('text', { ...labelAttrs, x: LC_LX + LC_LW / 2, y: LC_LY - 4, 'text-anchor': 'middle' });
  lLabelEl.textContent = 'L*';
  svg.appendChild(lLabelEl);

  // Dynamic group placeholders (cleared and redrawn by renderLabChart)
  const dotsGroup = svgEl('g', { id: 'lab-dots' });
  const ticksGroup = svgEl('g', { id: 'lab-ticks' });
  svg.appendChild(dotsGroup);
  svg.appendChild(ticksGroup);
}
```

```javascript
function renderLabChart(results) {
  const dotsGroup  = document.getElementById('lab-dots');
  const ticksGroup = document.getElementById('lab-ticks');
  // Clear dynamic elements
  while (dotsGroup.firstChild)  dotsGroup.removeChild(dotsGroup.firstChild);
  while (ticksGroup.firstChild) ticksGroup.removeChild(ticksGroup.firstChild);

  results.forEach(res => {
    const cfg   = ILLUMINANT_CONFIG[res.illuminantKey];
    const color = cfg.chipColor;
    const { L, a, b } = res.lab;

    // a*b* scatter dot
    const pt = abToSvg(a, b);
    dotsGroup.appendChild(svgEl('circle', {
      cx: pt.x, cy: pt.y, r: 5,
      fill: color
    }));

    // L* bar tick (14px wide, centered on bar)
    const ty = lToSvgY(L);
    ticksGroup.appendChild(svgEl('line', {
      x1: LC_LX - 1, y1: ty, x2: LC_LX + LC_LW + 1, y2: ty,
      stroke: color, 'stroke-width': '2'
    }));
  });
}
```
